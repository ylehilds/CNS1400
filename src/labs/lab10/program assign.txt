class Point
{
private:
	int xCoord;
	int yCoord;
public:
	Point( );
	Point(int, int);
	int getXCoord( )const;
	int getYCoord( )const;
	void setXCoord(int);
	void setYCoord(int);

{

	Point startPoint;
	Point endPoint;

	Line( );

	Point getEndPoint( )const;
	void setStartPoint(Point);
	void setEndPoint(Point);
};
